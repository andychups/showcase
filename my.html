    canvas = document.getElementById('js-video-canvas');
    context2d = canvas.getContext('2d'),
    timer = null;

    timer = setInterval(function() {

Проигрывание ролика без лагов


var video = document.getElementById('js-video'),
    controller = { ... };

video.addEventListener('progress', function() {
    var percent,
         /* Значение загруженной части видео (instance TimeRanges) */
        downloadedVideoPart = video.buffered.end(video.buffered.length-1),
         /* Значение продолжительности видео */
        videoDuration = video.duration;

    percent = downloadedVideoPart / videoDuration * 100;

    controller.progressBarUpdate(percent);

    if (percent === 100) {
        controller.showVideo();
    }
});

var video = document.getElementById('js-video'),
    someVideoProgress,
    controller = { ... };

someVideoProgress = setInterval(function() {
    var percent,
         /* Значение загруженной части видео (instance TimeRanges) */
        downloadedVideoPart = video.buffered.end(video.buffered.length-1),
        videoDuration = video.duration;

    if (typeof videoDuration === 'undefined') {
        return;
    }

    percent = downloadedVideoPart / videoDuration * 100;

    controller.progressBarUpdate(percent);

    if (percent === 100) {
        controller.showVideo();
        clearInterval(someVideoProgress);
    }
});
